Let's make a website health checker with Go. We'll use the `net/http` package to get the status of a URL.

```go
package main

import (
	"fmt"
	"net/http"
	"time"
)

func main() {
	urls := []string{
		"https://eshayat.com",
		"https://xenon.eshayat.com",
		"https://esyt.eshayat.com",
		"https://airpipe.eshayat.com",
	}

	start := time.Now()

	for _, url := range urls {
		res, err := http.Get(url)
		if err != nil {
			fmt.Println(url, ": down")
		} else {
			fmt.Println(url, ": up")
			res.Body.Close()
		}
	}

	fmt.Println("\ntime taken:", time.Since(start))
}
```
