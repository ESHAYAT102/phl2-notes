#### Creating the prepend method

```js
  constructor() {
    this.head = null;
    this.tail = null;
    this.length = 0;
  }

  append(...) {...}

  prepend(value) { // [!code highlight]

    if (this.head === null) { // [!code highlight]
      this.head = newNode; // [!code highlight]
      this.tail = newNode; // [!code highlight]
    } else {
    }
  }

  insert() {}
  remove() {}
  print() {}
}
```

Adding a node in beginning of the list:

```js
class LinkedList {
  constructor() {
    this.head = null;
    this.tail = null;
    this.length = 0;
  }

  append(...) {...}

  prepend(value) {
    const newNode = new Node(value);

    if (this.head === null) {
      this.head = newNode;
      this.tail = newNode;
    } else {
      newNode.next = this.head; // [!code highlight]
    }
  }

  insert() {}
  remove() {}
  print() {}
}
```

---

Identifying the new node as the head:

```js
class LinkedList {
  constructor() {
    this.head = null;
    this.tail = null;
    this.length = 0;
  }

  append(...) {...}

  prepend(value) {
    const newNode = new Node(value);

    if (this.head === null) {
      this.head = newNode;
      this.tail = newNode;
    } else {
      newNode.next = this.head;
      this.head = newNode; // [!code highlight]
    }

    this.length++; // [!code highlight]
  }

  insert() {}
  remove() {}
  print() {}
}
```

The code also updates the length.
