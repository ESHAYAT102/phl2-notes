## Problem

Create a program that:

- Shows a menu with options:
  - 1. Calculate grade
  - 2. Check pass/fail status
  - 3. Exit
- Use a switch system for menu selection
- For option 1: Take a score between 1 to 100 and show a text as per their grade
- For option 2: Use a switch to check if a score is pass or fail

---

## My Solution

```go title="main.go"
package main

import "fmt"

func checkPass(score int) bool {
	if score >= 40 {
		return true
	}
	return false
}

func checkGrade(score int) string {
	switch {
	case score < 40:
		return "F"
	case score < 50:
		return "E"
	case score < 60:
		return "D"
	case score < 70:
		return "C"
	case score < 80:
		return "B"
	case score < 90:
		return "A"
	case score < 100:
		return "A+"
	}
	return "invalid"
}

func main() {
	fmt.Println("1) Calculate Grade")
	fmt.Println("2) Check pass/fail status")
	fmt.Println("3) Exit")
	fmt.Print("Enter your choice: ")
	var choice int
	fmt.Scan(&choice)
	if choice == 1 {
	}
	switch choice {
	case 1:
		var grade int
		fmt.Print("Enter grade: ")
		fmt.Scan(&grade)
		fmt.Printf("Your grade is: %s\n", checkGrade(grade))
	case 2:
		var check int
		fmt.Print("Enter grade: ")
		fmt.Scan(&check)
		if checkPass(check) {
			fmt.Println("You passed")
			break
		}
		fmt.Println("You failed")
	case 3:
		fmt.Println("Exiting...")
	default:
		fmt.Println("Invalid input")
	}
}
```

## Best Solution

```go title="main.go"
package main

import "fmt"

func checkPass(score int) bool {
	return score >= 40
}

func checkGrade(score int) string {
	if score < 0 || score > 99 {
		return "invalid"
	}
	grades := []string{"F", "F", "F", "F", "E", "D", "C", "B", "A", "A+"}
	return grades[score/10]
}

func main() {
	fmt.Println("1) Calculate Grade")
	fmt.Println("2) Check pass/fail status")
	fmt.Println("3) Exit")
	fmt.Print("Enter your choice: ")
	var choice int
	fmt.Scan(&choice)
	switch choice {
	case 1:
		var grade int
		fmt.Print("Enter grade: ")
		fmt.Scan(&grade)
		fmt.Printf("Your grade is: %s\n", checkGrade(grade))
	case 2:
		var check int
		fmt.Print("Enter grade: ")
		fmt.Scan(&check)
		if checkPass(check) {
			fmt.Println("You passed")
		} else {
			fmt.Println("You failed")
		}
	case 3:
		fmt.Println("Exiting...")
	default:
		fmt.Println("Invalid input")
	}
}
```
