## Function guard

```ts
class Person {
  constructor(public name: string) {}
}

class Developer extends Person {
  constructor(name: string) {
    super(name);
  }

  code(hours: number) {
    console.log(`${this.name} codes ${hours} hours a day.`);
  }
}

class Designer extends Person {
  constructor(name: string) {
    super(name);
  }

  design(hours: number) {
    console.log(`${this.name} designs ${hours} hours a day.`);
  }
}

const dev = new Developer("Eshayat");

dev.code(4);

const isDeveloper = (npc: Person) => {
  return npc instanceof Developer;
};

const isDesigner = (npc: Person) => {
  return npc instanceof Designer;
};

const isUnknown = (npc: Person | Developer | Designer) => {
  return "hell nahhh";
};

console.log(isDeveloper(dev));
```
