Function with return:

```go
package main

import "fmt"

func coffee(coffeeType string) string { // [!code highlight]
	coffeeToReturn := fmt.Sprintf("%v Coffee\n", coffeeType)
	return coffeeToReturn // [!code highlight]
}

func main() {
	returnedCoffee := coffee("Africano") // [!code highlight]
	fmt.Print("I'm Having ", returnedCoffee) // [!code highlight]
}
```
